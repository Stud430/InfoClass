

CREATE TABLE `cursos` (
  `id` int(11) AUTO_INCREMENT PRIMARY KEY,
  `curso` varchar(50) NOT NULL,
  `plataforma` varchar(30) NOT NULL,
  `duracao` varchar(25),
  `endereco` varchar(150) NOT NULL,
  `inicio` varchar(10),
  `termino` varchar(10),
  `usuario` varchar(25),
  `senha` varchar(15),
  `status_curso` varchar(15),
  `created` varchar(10)
)
